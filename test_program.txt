// Testovaci program
// Obsahuje vsechny pozadovane instrukce:
// add, sub, mul, div, mod, uminus, concat, and, or, gt, lt, eq, not, itof, push, pop, load, save, label, jmp, fjmp, print, read

int i, j, k;
float x, y, z;
bool b1, b2;
string s1, s2;

// 1. Test cteni a zapisu (read T, print n, push T, save id)
write "Zadejte cele cislo (i), desetinne (x), bool (b1) a retezec (s1):";
read i, x, b1, s1;

// 2. Test celociselne aritmetiky (add I, sub I, mul I, div I, mod, uminus I)
j = 10;
k = i + j;
k = i - j;
k = i * j;
k = i / j;
k = i % 3;
k = -i;

// 3. Test desetinne aritmetiky (add F, sub F, mul F, div F, uminus F)
y = 2.5;
z = x + y;
z = x - y;
z = x * y;
z = x / y;
z = -x;

// 4. Test automatickeho pretypovani (itof)
z = i + x;  // i se prevede na float
z = 5;      // int literal se prevede na float pri prirazeni

// 5. Test retezcu (concat, eq S)
s2 = s1 . " - test";
b2 = s1 == "konec";

// 6. Test logiky a porovnavani (and, or, not, gt, lt, eq I/F)
b2 = (i > 0) && (x < 100.0);
b2 = b2 || (i == j);
b2 = !b1;

// 7. Test rizeni toku (label, jmp, fjmp)
if (i > 10) {
    write "i je vetsi nez 10";
} else {
    write "i je 10 nebo mene";
}

k = 0;
while (k < 3) {
    write "Iterace cyklu:", k;
    k = k + 1;
}

int f;
for (f = 0; f < 5; f = f + 1) {
    write "For cyklus - iterace:", f;
}

// 8. Test odstraneni hodnoty ze zasobniku (pop)
i + 1; // Vysledek se nevyuzije, zavola se pop

float testTer;
testTer = (i > 10) ? 1.1 : 2;
write "Ternarni operater vysledek:", testTer;

// 9. Test indexace stringu (indexingExpr, get_char)
string ind;
ind = "Abeceda";
write "Pismena ze stringu 'Abeceda' na pozici 0 a ", i , ":", ind[0], " ", ind[i];

// 10. Test FILE operaci (fopen, fappend)
FILE soubor;
fopen soubor, "vystup_testu.txt";
fappend soubor, "Vysledek testu: ", s2, ", i=", i, ", b2=", b2, "\n";
write "Zapis do souboru 'vystup_testu.txt' hotov.";

write "Test dokoncen:", s2, " ", b2;
